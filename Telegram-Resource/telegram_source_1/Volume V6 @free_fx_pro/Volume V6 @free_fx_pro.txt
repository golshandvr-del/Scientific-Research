//------------------------------------------------------------------------------
// Helpers
//------------------------------------------------------------------------------
f_table_pos(pos) =>
    switch pos
        "Top Left" => position.top_left
        "Top Right" => position.top_right
        "Bottom Left" => position.bottom_left
        => position.bottom_right

f_table_size(sizeText) =>
    switch sizeText
        "Tiny" => size.tiny
        "Small" => size.small
        "Normal" => size.normal
        => size.large

f_clamp(v, lo, hi) =>
    math.max(lo, math.min(hi, v))

f_format_compact(v) =>
    absV = math.abs(v)
    result = ""
    if absV >= 1e12
        result := str.tostring(v / 1e12, "#.##") + "T"
    else if absV >= 1e9
        result := str.tostring(v / 1e9, "#.##") + "B"
    else if absV >= 1e6
        result := str.tostring(v / 1e6, "#.##") + "M"
    else if absV >= 1e3
        result := str.tostring(v / 1e3, "#.##") + "K"
    else
        result := str.tostring(v, "#.##")
    result

f_format_money(v, unitMode) =>
    divisor = unitMode == "Millions" ? 1000000.0 : unitMode == "Crores" ? 10000000.0 : 1.0
    suffix = unitMode == "Millions" ? "M" : unitMode == "Crores" ? "Cr" : ""
    unitMode == "Raw" ? f_format_compact(v) : str.tostring(v / divisor, "#.##") + suffix

f_highest_down_volume(requiredDownDays, maxScan) =>
    float highest = na
    int found = 0
    for i = 1 to maxScan
        if i < bar_index
            isDownBar = close[i] < close[i + 1]
            if isDownBar
                highest := na(highest) ? volume[i] : math.max(highest, volume[i])
                found += 1
                if found >= requiredDownDays
                    break
    highest

f_sum_up_down_volume(length) =>
    float upSum = 0.0
    float downSum = 0.0
    for i = 1 to length
        if i < bar_index
            if close[i - 1] > close[i]
                upSum += volume[i - 1]
            else if close[i - 1] < close[i]
                downSum += volume[i - 1]
    [upSum, downSum]

f_close_position() =>
    rng = high - low
    rng > 0 ? ((close - low) / rng) * 100.0 : 50.0

f_close_in_upper_zone(thresholdPct) =>
    f_close_position() >= thresholdPct

f_close_in_top_zone(topPct) =>
    100.0 - f_close_position() <= topPct

isUpDay = close > close[1]
isDownDay = close < close[1]
volAvg = ta.sma(volume, avgLen)
relativeVol = volAvg > 0 ? volume / volAvg : na
relativeVolPct = relativeVol * 100.0
avgDollarVol = ta.sma(volume * close, statsLookback)

[upVolSum, downVolSum] = f_sum_up_down_volume(statsLookback)
upDownRatio = downVolSum > 0 ? upVolSum / downVolSum : na

shortMa = ta.sma(close, ppvShortMaLen)
longMa = ta.sma(close, ppvLongMaLen)
trendOk = close > shortMa and close > longMa

maxScanBars = math.max(ppvDownDays * 20, 100)
highestRecentDownVol = f_highest_down_volume(ppvDownDays, maxScanBars)

ppvBase = isUpDay and not na(highestRecentDownVol) and volume > highestRecentDownVol
ppvUpperZoneOk = not ppvRequireCloseNearHigh or f_close_in_upper_zone(ppvUpperRangePct)
ppvShortMaOk = not ppvRequireAboveShortMa or close > shortMa
ppvLongMaOk = not ppvRequireAboveLongMa or close > longMa
isPPV = ppvBase and ppvUpperZoneOk and ppvShortMaOk and ppvLongMaOk

isStrongUp = isUpDay and volume > volAvg and f_close_position() >= 50
isStrongDown = isDownDay and volume > volAvg and f_close_position() <= 50
isDryVolume = volume < volAvg * dryFraction

bullSnortVolAvg = ta.sma(volume, bullSnortVolLen)
bullSnortVolumeOk = bullSnortVolAvg > 0 and volume >= bullSnortVolAvg * bullSnortVolMult
bullSnortCloseZoneOk = f_close_in_top_zone(bullSnortTopZonePct)
bullSnortCloseOk = close > close[1]
isBullSnort = showBullSnort and bullSnortVolumeOk and bullSnortCloseZoneOk and bullSnortCloseOk

//------------------------------------------------------------------------------
// Accumulation / distribution pressure
//------------------------------------------------------------------------------
closeLocationValue = high != low ? ((close - low) - (high - close)) / (high - low) : 0.0
effort = f_clamp(nz(relativeVol, 1.0), 0.0, 4.0)
volumePressure = closeLocationValue * effort
pressureFast = ta.ema(volumePressure, math.max(2, math.round(pressureLen / 3)))
pressureSlow = ta.ema(volumePressure, pressureLen)
moneyFlowVolume = closeLocationValue * volume
volumeSum = math.sum(volume, pressureLen)
moneyFlowSum = math.sum(moneyFlowVolume, pressureLen)
cmf = volumeSum > 0 ? moneyFlowSum / volumeSum : 0.0
obv = ta.cum(math.sign(ta.change(close)) * volume)
obvRising = ta.ema(obv, 5) > ta.ema(obv, 13)
obvFalling = ta.ema(obv, 5) < ta.ema(obv, 13)

distributionDay = showDistribution and isDownDay and relativeVol >= 1.2 and f_close_position() <= 45
upthrustSell = showDistribution and high > high[1] and close < open and f_close_position() <= 45 and relativeVol >= 1.5
heavyChurn = showDistribution and relativeVol >= 2.0 and math.abs(close - open) <= (high - low) * 0.25 and f_close_position() < 55
sellPressure = distributionDay or upthrustSell or heavyChurn

constructiveDryPullback = showPullbackAdds and isDryVolume and close >= longMa and close <= shortMa * 1.03 and close >= longMa * 0.97
demandBar = isStrongUp and f_close_position() >= 65 and relativeVol >= 1.25

rawScore = 0.0
rawScore += isPPV ? 28 : 0
rawScore += isBullSnort ? 22 : 0
rawScore += demandBar ? 16 : 0
rawScore += constructiveDryPullback ? 12 : 0
rawScore += pressureFast > pressureSlow ? 10 : -10
rawScore += cmf > 0.05 ? 12 : cmf < -0.05 ? -12 : 0
rawScore += obvRising ? 8 : obvFalling ? -8 : 0
rawScore += trendOk ? 10 : close < longMa ? -10 : 0
rawScore -= distributionDay ? 24 : 0
rawScore -= upthrustSell ? 22 : 0
rawScore -= heavyChurn ? 16 : 0
rawScore := f_clamp(rawScore, -100, 100)
accumScore = ta.ema(rawScore, scoreSmoothLen)

buyThreshold = signalMode == "Aggressive" ? 42 : signalMode == "Balanced" ? 52 : 62
sellThreshold = signalMode == "Aggressive" ? -38 : signalMode == "Balanced" ? -48 : -58
addThreshold = signalMode == "Aggressive" ? 22 : signalMode == "Balanced" ? 30 : 38

buySignal = accumScore >= buyThreshold and (isPPV or isBullSnort or demandBar) and close > shortMa
addSignal = showPullbackAdds and accumScore >= addThreshold and constructiveDryPullback
sellSignal = showDistribution and accumScore <= sellThreshold and sellPressure
trimSignal = showDistribution and sellPressure and accumScore < 10 and not sellSignal

verdict = accumScore >= buyThreshold ? "BUY / ACCUMULATE" : accumScore >= addThreshold ? "ADD ON DRY-UP" : accumScore <= sellThreshold ? "SELL / DEFEND" : accumScore < -15 ? "DISTRIBUTION" : "NEUTRAL"
verdictColor = accumScore >= buyThreshold ? colorBuy : accumScore >= addThreshold ? colorAdd : accumScore <= sellThreshold ? colorSell : accumScore < -15 ? color.new(colorSell, 15) : colorNeutral
