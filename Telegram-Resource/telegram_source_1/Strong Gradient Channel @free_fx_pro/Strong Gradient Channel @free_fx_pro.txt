ramp5(float t) =>
    float u = math.max(0.0, math.min(1.0, t)) * 4.0
    int i = math.min(3, int(u))
    color a = i == 0 ? r0 : i == 1 ? r1 : i == 2 ? r2 : r3
    color b = i == 0 ? r1 : i == 1 ? r2 : i == 2 ? r3 : r4
    color.from_gradient(u - i, 0.0, 1.0, a, b)
