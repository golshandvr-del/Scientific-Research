// ─────────────────────────────────────────────────────────────────────────

f_pos(string s) =>
    switch s
        "Top Right"    => position.top_right
        "Top Left"     => position.top_left
        "Bottom Right" => position.bottom_right
        "Bottom Left"  => position.bottom_left
        => position.middle_right

hudTS = hudSize == "Tiny" ? size.tiny : hudSize == "Small" ? size.small : hudSize == "Large" ? size.large : size.normal

atr = ta.atr(atrLenT)
atrSafe = na(atr) or atr <= 0 ? syminfo.mintick : atr
var table wm = table.new(position.bottom_right, 1, 1)
if barstate.isfirst
    table.cell(wm, 0, 0, "Telegram @free_fx_pro", text_color = color.new(color.gray, 70), text_size = size.large)
// ─────────────────────────────────────────────────────────────────────────
//  KILLZONES  (session windows in a chosen timezone)
// ─────────────────────────────────────────────────────────────────────────
inLdn = kzLdnOn and not na(time(timeframe.period, kzLdn, kzTz))
inNy  = kzNyOn  and not na(time(timeframe.period, kzNy,  kzTz))
inKZ  = inLdn or inNy
killOK = not useKZ or inKZ

// ─────────────────────────────────────────────────────────────────────────
//  SWING PIVOTS  (confirmed pivLen bars back — non-repainting)
// ─────────────────────────────────────────────────────────────────────────
ph = ta.pivothigh(high, pivLen, pivLen)
pl = ta.pivotlow(low,  pivLen, pivLen)

// active (most recent) swing levels + a "broken" flag so each fires ONCE
var float shPrice = na
var int   shBar   = na
var bool  shBroken= false
var float slPrice = na
var int   slBar   = na
var bool  slBroken= false

// two most recent swing extremes (for SMT) + correlated readings
var float lastSHp = na
var float prevSHp = na
var float lastSLp = na
var float prevSLp = na

// ─────────────────────────────────────────────────────────────────────────
//  SMT — correlated instrument (lookahead OFF, blank = self = disabled)
// ─────────────────────────────────────────────────────────────────────────
smtSym = smtInput == "" ? syminfo.tickerid : smtInput
[cHigh, cLow] = request.security(smtSym, timeframe.period, [high, low], lookahead = barmerge.lookahead_off)
var float lastSHc = na
var float prevSHc = na
var float lastSLc = na
var float prevSLc = na

if not na(ph)
    shPrice := ph
    shBar   := bar_index - pivLen
    shBroken:= false
    prevSHp := lastSHp
    lastSHp := ph
    prevSHc := lastSHc
    lastSHc := cHigh[pivLen]
if not na(pl)
    slPrice := pl
    slBar   := bar_index - pivLen
    slBroken:= false
    prevSLp := lastSLp
    lastSLp := pl
    prevSLc := lastSLc
    lastSLc := cLow[pivLen]

// SMT: our new extreme is unconfirmed by the correlated symbol
smtHasSym = smtInput != ""
bearSMT = smtHasSym and not na(prevSHp) and lastSHp > prevSHp and lastSHc <= prevSHc  // we higher high, corr didn't
bullSMT = smtHasSym and not na(prevSLp) and lastSLp < prevSLp and lastSLc >= prevSLc  // we lower low, corr didn't

if showSwn and not na(ph)
    label.new(bar_index - pivLen, ph, "·", style = label.style_none, textcolor = color.new(MGREY, 30), size = size.tiny)
if showSwn and not na(pl)
    label.new(bar_index - pivLen, pl, "·", style = label.style_none, textcolor = color.new(MGREY, 30), size = size.tiny)

if showSMT and bearSMT
    label.new(bar_index - pivLen, ph, "SMT", style = label.style_label_down, color = color.new(ACC_DN, 20), textcolor = WHITE, size = size.tiny, yloc = yloc.abovebar)
if showSMT and bullSMT
    label.new(bar_index - pivLen, pl, "SMT", style = label.style_label_up, color = color.new(ACC_UP, 20), textcolor = WHITE, size = size.tiny, yloc = yloc.belowbar)

// ─────────────────────────────────────────────────────────────────────────
//  MARKET STRUCTURE  (BOS / CHoCH — fires ONCE per level, then that level is
//  consumed until a fresh swing forms; this is the fix for sprayed labels)
// ─────────────────────────────────────────────────────────────────────────
var int bias = 0

bullBreak = not na(shPrice) and not shBroken and close > shPrice
bearBreak = not na(slPrice) and not slBroken and close < slPrice

bool choChUp = false
bool choChDn = false

if bullBreak
    choChUp := bias < 0          // break against a bearish bias = change of character
    shBroken := true
    bias := 1
if bearBreak
    choChDn := bias > 0
    slBroken := true
    bias := -1

// draw the broken level cleanly: a line from the swing to the break + a label
if showBos and (bullBreak or bearBreak)
    lvl = bullBreak ? shPrice : slPrice
    x1  = bullBreak ? shBar : slBar
    line.new(x1, lvl, bar_index, lvl, color = bullBreak ? color.new(ACC_UP, 30) : color.new(ACC_DN, 30), style = line.style_dashed, width = 1)
    txt = (bullBreak ? (choChUp ? "CHoCH" : "BOS") : (choChDn ? "CHoCH" : "BOS"))
    label.new(bar_index, lvl, txt, style = bullBreak ? label.style_label_up : label.style_label_down, color = bullBreak ? color.new(ACC_UP, 15) : color.new(ACC_DN, 15), textcolor = WHITE, size = size.tiny, yloc = bullBreak ? yloc.belowbar : yloc.abovebar)

// displacement (used to qualify an MSS)
dispUp = (close - open) > dispAtr * atrSafe
dispDn = (open - close) > dispAtr * atrSafe

// ─────────────────────────────────────────────────────────────────────────
//  LIQUIDITY LEVELS + SWEEPS
// ─────────────────────────────────────────────────────────────────────────
var line bslLine = na
var line sslLine = na
var label bslLbl = na
var label sslLbl = na
if showLiq and barstate.islast
    if na(bslLine)
        bslLine := line.new(bar_index, na, bar_index, na, color = color.new(MGREY, 40), style = line.style_dotted)
        sslLine := line.new(bar_index, na, bar_index, na, color = color.new(MGREY, 40), style = line.style_dotted)
        bslLbl  := label.new(bar_index, na, "", style = label.style_label_left, color = color.new(INK, 100), textcolor = MUTE, size = size.tiny)
        sslLbl  := label.new(bar_index, na, "", style = label.style_label_left, color = color.new(INK, 100), textcolor = MUTE, size = size.tiny)
    if not na(shPrice)
        line.set_xy1(bslLine, bar_index - 40, shPrice)
        line.set_xy2(bslLine, bar_index + 8, shPrice)
        label.set_xy(bslLbl, bar_index + 8, shPrice)
        label.set_text(bslLbl, " BSL " + str.tostring(shPrice, format.mintick))
    if not na(slPrice)
        line.set_xy1(sslLine, bar_index - 40, slPrice)
        line.set_xy2(sslLine, bar_index + 8, slPrice)
        label.set_xy(sslLbl, bar_index + 8, slPrice)
        label.set_text(sslLbl, " SSL " + str.tostring(slPrice, format.mintick))

// sweep = raid a swing then close back inside (failed break)
sweepUp = not na(shPrice) and high > shPrice and close < shPrice   // raid buy-side -> bearish intent
sweepDn = not na(slPrice) and low  < slPrice and close > slPrice   // raid sell-side -> bullish intent

if showSwp and sweepUp
    label.new(bar_index, high, "sweep", style = label.style_label_down, color = color.new(ACC_DN, 35), textcolor = WHITE, size = size.tiny, yloc = yloc.abovebar)
if showSwp and sweepDn
    label.new(bar_index, low, "sweep", style = label.style_label_up, color = color.new(ACC_UP, 35), textcolor = WHITE, size = size.tiny, yloc = yloc.belowbar)

// ─────────────────────────────────────────────────────────────────────────
//  ORDER BLOCKS  (last opposing candle before a displacement leg)
// ─────────────────────────────────────────────────────────────────────────
var array<box> obBox = array.new<box>()
var array<int> obDir = array.new<int>()

bullOB = dispUp and close[1] < open[1]
bearOB = dispDn and close[1] > open[1]

if showOB and bullOB
    b = box.new(bar_index - 1, math.max(open[1], close[1]), bar_index + 6, low[1], border_color = color.new(ACC_UP, 45), bgcolor = color.new(ACC_UP, 88), border_width = 1, text = "Demand", text_color = color.new(ACC_UP, 20), text_size = size.tiny, text_halign = text.align_left, text_valign = text.align_bottom)
    array.push(obBox, b)
    array.push(obDir, 1)
if showOB and bearOB
    b = box.new(bar_index - 1, high[1], bar_index + 6, math.min(open[1], close[1]), border_color = color.new(ACC_DN, 45), bgcolor = color.new(ACC_DN, 88), border_width = 1, text = "Supply", text_color = color.new(ACC_DN, 20), text_size = size.tiny, text_halign = text.align_left, text_valign = text.align_top)
    array.push(obBox, b)
    array.push(obDir, -1)

if array.size(obBox) > 0
    for i = array.size(obBox) - 1 to 0
        b   = array.get(obBox, i)
        d   = array.get(obDir, i)
        top = box.get_top(b)
        bot = box.get_bottom(b)
        mitig = d == 1 ? close < bot : close > top
        if mitig
            box.delete(b)
            array.remove(obBox, i)
            array.remove(obDir, i)
        else
            box.set_right(b, bar_index + 6)
while array.size(obBox) > obMax
    box.delete(array.shift(obBox))
    array.shift(obDir)

// ─────────────────────────────────────────────────────────────────────────
//  FAIR VALUE GAPS  (3-bar imbalance) — tracked so the MSS can adopt one
// ─────────────────────────────────────────────────────────────────────────
var array<box>   fvgBox = array.new<box>()
var array<int>   fvgDir = array.new<int>()

bullFVG = low > high[2] and (low - high[2]) > fvgMinAtr * atrSafe
bearFVG = high < low[2] and (low[2] - high) > fvgMinAtr * atrSafe

if showFVG and bullFVG
    b = box.new(bar_index - 2, low, bar_index + 3, high[2], border_color = color.new(GOLD, 60), bgcolor = color.new(GOLD, 90))
    array.push(fvgBox, b)
    array.push(fvgDir, 1)
if showFVG and bearFVG
    b = box.new(bar_index - 2, low[2], bar_index + 3, high, border_color = color.new(GOLD, 60), bgcolor = color.new(GOLD, 90))
    array.push(fvgBox, b)
    array.push(fvgDir, -1)

if array.size(fvgBox) > 0
    for i = array.size(fvgBox) - 1 to 0
        b   = array.get(fvgBox, i)
        d   = array.get(fvgDir, i)
        top = box.get_top(b)
        bot = box.get_bottom(b)
        filled = d == 1 ? close < bot : close > top
        if filled
            box.delete(b)
            array.remove(fvgBox, i)
            array.remove(fvgDir, i)
        else
            box.set_right(b, bar_index + 3)
while array.size(fvgBox) > fvgMax
    box.delete(array.shift(fvgBox))
    array.shift(fvgDir)

// ─────────────────────────────────────────────────────────────────────────
//  THE MODEL STATE MACHINE  (sweep -> MSS -> entry-zone tap)
//  stage 0 = idle, 1 = swept/armed, 2 = MSS confirmed & waiting for the retrace
// ─────────────────────────────────────────────────────────────────────────
var int   bStage = 0            // bullish path
var float bSweepLo= na          // swept extreme (SL basis)
var int   bStageBar= na
var float bZoneTop = na
var float bZoneBot = na

var int   sStage = 0            // bearish path
var float sSweepHi= na
var int   sStageBar= na
var float sZoneTop = na
var float sZoneBot = na

// 1) sweep arms the path
if sweepDn
    bStage    := 1
    bSweepLo  := low
    bStageBar := bar_index
if sweepUp
    sStage    := 1
    sSweepHi  := high
    sStageBar := bar_index

// 2) MSS (displacement structure break in the reversal direction) confirms
mssUp = bullBreak and dispUp
mssDn = bearBreak and dispDn

if bStage == 1 and mssUp and (bar_index - bStageBar) <= mssWin
    bStage    := 2
    bStageBar := bar_index
    eq = (high + bSweepLo) / 2.0                    // 50% equilibrium of the reversal leg
    // adopt a fresh bullish FVG if one just formed, else an equilibrium band
    if bullFVG
        bZoneTop := low
        bZoneBot := high[2]
    else
        bZoneTop := eq + zonePad * atrSafe
        bZoneBot := eq - zonePad * atrSafe
if sStage == 1 and mssDn and (bar_index - sStageBar) <= mssWin
    sStage    := 2
    sStageBar := bar_index
    eq = (low + sSweepHi) / 2.0
    if bearFVG
        sZoneTop := low[2]
        sZoneBot := high
    else
        sZoneTop := eq + zonePad * atrSafe
        sZoneBot := eq - zonePad * atrSafe

// expire stale setups
if bStage == 1 and (bar_index - bStageBar) > mssWin
    bStage := 0
if sStage == 1 and (bar_index - sStageBar) > mssWin
    sStage := 0
if bStage == 2 and (bar_index - bStageBar) > entryWin
    bStage := 0
if sStage == 2 and (bar_index - sStageBar) > entryWin
    sStage := 0

// 3) entry = price retraces into the zone with a reaction, in a killzone,
//    optionally confirmed by SMT
bullTap = bStage == 2 and low <= bZoneTop and close >= bZoneBot and close >= open
bearTap = sStage == 2 and high >= sZoneBot and close <= sZoneTop and close <= open

buy  = bullTap and killOK and (not useSMT or bullSMT)
sell = bearTap and killOK and (not useSMT or bearSMT)

// consume the setup on entry
if buy
    bStage := 0
if sell
    sStage := 0

// mutual exclusivity + debounce (never overlap on any timeframe)
sell := sell and not buy
var int lastSigBar = na
okGap = na(lastSigBar) or (bar_index - lastSigBar) >= minGap
buy  := buy  and okGap
sell := sell and okGap
sig  = buy or sell
if sig
    lastSigBar := bar_index

// ─────────────────────────────────────────────────────────────────────────
//  TP/SL BOXES  (red RISK entry->SL, green REWARD entry->TP3, labelled edges)
//  SL rests BEYOND the swept liquidity — the TJR invalidation.
// ─────────────────────────────────────────────────────────────────────────
var float sEntry = na
var float sSL    = na
var float sT1    = na
var float sT2    = na
var float sT3    = na
var int   sDir   = 0
var bool  sOpen  = false
var int   wins   = 0
var int   losses = 0

if sOpen
    if sDir == 1
        if high >= sT1
            wins   += 1
            sOpen  := false
        else if low <= sSL
            losses += 1
            sOpen  := false
    else
        if low <= sT1
            wins   += 1
            sOpen  := false
        else if high >= sSL
            losses += 1
            sOpen  := false

var box   riskBox = na
var box   rewBox  = na
var line  eLine   = na
var line  slLine  = na
var line  t1Line  = na
var line  t2Line  = na
var label lbE = na
var label lbS = na
var label lb1 = na
var label lb2 = na
var label lb3 = na

f_setLbl(label lb, float p, string txt, color c) =>
    label.set_xy(lb, bar_index + boxW, p)
    label.set_text(lb, txt)
    label.set_textcolor(lb, c)

if sig
    if sOpen
        losses += 1
        sOpen  := false
    sDir   := buy ? 1 : -1
    slBase  = buy ? (na(bSweepLo) ? low : bSweepLo) - slPad * atrSafe : (na(sSweepHi) ? high : sSweepHi) + slPad * atrSafe
    sEntry := close
    risk    = math.abs(sEntry - slBase)
    risk    := risk > 0 ? risk : atrSafe
    sSL    := buy ? sEntry - risk : sEntry + risk
    sT1    := buy ? sEntry + risk * tp1R : sEntry - risk * tp1R
    sT2    := buy ? sEntry + risk * tp2R : sEntry - risk * tp2R
    sT3    := buy ? sEntry + risk * tp3R : sEntry - risk * tp3R
    sOpen  := true

    if showBox
        box.delete(riskBox)
        box.delete(rewBox)
        riskBox := box.new(bar_index, sEntry, bar_index + boxW, sSL, bgcolor = color.new(ACC_DN, 86), border_color = color.new(ACC_DN, 40), border_width = 1)
        rewBox  := box.new(bar_index, sEntry, bar_index + boxW, sT3, bgcolor = color.new(ACC_UP, 88), border_color = color.new(ACC_UP, 40), border_width = 1)
        if na(eLine)
            eLine  := line.new(bar_index, sEntry, bar_index + boxW, sEntry, color = color.new(GOLD, 10),   width = 1)
            slLine := line.new(bar_index, sSL,    bar_index + boxW, sSL,    color = color.new(ACC_DN, 20),  width = 1, style = line.style_dashed)
            t1Line := line.new(bar_index, sT1,    bar_index + boxW, sT1,    color = color.new(ACC_UP, 30),  width = 1, style = line.style_dotted)
            t2Line := line.new(bar_index, sT2,    bar_index + boxW, sT2,    color = color.new(ACC_UP, 30),  width = 1, style = line.style_dotted)
            lbE := label.new(bar_index + boxW, sEntry, "", style = label.style_label_left, color = color.new(WHITE, 0), textcolor = INK, size = size.small)
            lbS := label.new(bar_index + boxW, sSL,    "", style = label.style_label_left, color = color.new(WHITE, 0), textcolor = INK, size = size.small)
            lb1 := label.new(bar_index + boxW, sT1,    "", style = label.style_label_left, color = color.new(WHITE, 0), textcolor = INK, size = size.small)
            lb2 := label.new(bar_index + boxW, sT2,    "", style = label.style_label_left, color = color.new(WHITE, 0), textcolor = INK, size = size.small)
            lb3 := label.new(bar_index + boxW, sT3,    "", style = label.style_label_left, color = color.new(WHITE, 0), textcolor = INK, size = size.small)
        line.set_xy1(eLine,  bar_index, sEntry)
        line.set_xy2(eLine,  bar_index + boxW, sEntry)
        line.set_xy1(slLine, bar_index, sSL)
        line.set_xy2(slLine, bar_index + boxW, sSL)
        line.set_xy1(t1Line, bar_index, sT1)
        line.set_xy2(t1Line, bar_index + boxW, sT1)
        line.set_xy1(t2Line, bar_index, sT2)
        line.set_xy2(t2Line, bar_index + boxW, sT2)
        f_setLbl(lbE, sEntry, " ENTRY " + str.tostring(sEntry, format.mintick), INK)
        f_setLbl(lbS, sSL,    " SL " + str.tostring(sSL, format.mintick), ACC_DN)
        f_setLbl(lb1, sT1,    " TP1 · " + str.tostring(tp1R, "#.#") + "R " + str.tostring(sT1, format.mintick), ACC_UP)
